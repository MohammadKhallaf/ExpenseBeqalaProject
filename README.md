1. python3 -m venv venv
2. source venv/bin/activate
3. pip install <br /> django
               <br /> djangorestframework
               <br /> djangorestframework-simplejwt==4.8
               <br /> djoser 
               <br /> psycopg2 
               <br /> psycopg2-binary

5. python manage.py migrate
